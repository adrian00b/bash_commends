cd  - change directory
	cd ~ - home directory

.bashrc - plik w ktorym znajduja sie rozne aliasy, mozemy je wiec dodawac po otworzeniu np uzywajac subl .bashrc

pwd - print working directory

mkdir - make direction (stworz folder) 

ls - wypisuje zawartosc katalogu

echo - wypisuje w konsoli tekst
	echo 'text' > 'nazwa_pliku'  - wpisuje tekst do pliku
	UWAGA! w przypadku gdy plik nie istnieje, zostanie stworzony, ale gdy istnieje JEGO ZAWARTOSC ZOSTANIE PODMIENIONA

cat '' - wyswietlenie zawartosci pliku w wierszu polecen
	-n numeruje linie 

mv 'file1' 'file2' - change name of file1 to file2
mv 'directory' 'directory' - mozna przeniesc katalog albo plik z jednego directory do drugiego
